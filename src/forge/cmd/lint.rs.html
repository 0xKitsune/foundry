11/// CLI arguments for `forge lint`.
12#[derive(Clone, Debug, Parser)]
13pub struct LintArgs {
14    /// The project's root path.
15    ///
16    /// By default root of the Git repository, if in one,
17    /// or the current working directory.
18    #[arg(long, value_hint = ValueHint::DirPath, value_name = "PATH")]
19    root: Option<PathBuf>,
20
21    /// Include only the specified files when linting.
22    #[arg(long, value_hint = ValueHint::FilePath, value_name = "FILES", num_args(1..))]
23    include: Option<Vec<PathBuf>>,
24
25    /// Exclude the specified files when linting.
26    #[arg(long, value_hint = ValueHint::FilePath, value_name = "FILES", num_args(1..))]
27    exclude: Option<Vec<PathBuf>>,
28
29    /// Specifies which lints to run based on severity.
30    ///
31    /// Supported values: `high`, `med`, `low`, `info`, `gas`.
32    #[arg(long, value_name = "SEVERITY", num_args(1..))]
33    severity: Option<Vec<Severity>>,
34}
35
36impl_figment_convert_basic!(LintArgs);
37
38impl LintArgs {
39    pub fn run(self) -> Result<()> {
40        let config = self.load_config()?;
41        let project = config.project()?;
42
43        // Get all source files from the project
44        let mut sources =
45            project.paths.read_input_files()?.keys().cloned().collect::<Vec<PathBuf>>();
46
47        // Add included paths to sources
48        if let Some(include_paths) = &self.include {
49            let included =
50                include_paths.iter().filter(|path| path.exists()).cloned().collect::<Vec<_>>();
51            sources.extend(included);
52        }
53
54        // Remove excluded files from sources
55        if let Some(exclude_paths) = &self.exclude {
56            let excluded = exclude_paths.iter().cloned().collect::<HashSet<_>>();
57            sources.retain(|path| !excluded.contains(path));
58        }
59
60        if sources.is_empty() {
61            sh_println!("Nothing to lint")?;
62            std::process::exit(0);
63        }
64
65        if project.compiler.solc.is_some() {
66            SolidityLinter::new().with_severity(self.severity).lint(&sources);
67        } else {
68            todo!("Linting not supported for this language");
69        };
70
71        Ok(())
72    }
73}

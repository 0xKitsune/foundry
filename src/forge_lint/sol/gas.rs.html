7impl<'ast> Visit<'ast> for AsmKeccak256 {
8    type BreakValue = ();
9
10    fn visit_expr(&mut self, expr: &'ast Expr<'ast>) -> ControlFlow<Self::BreakValue> {
11        if let ExprKind::Call(expr, _) = &expr.kind {
12            if let ExprKind::Ident(ident) = &expr.kind {
13                if ident.name.as_str() == "keccak256" {
14                    self.results.push(expr.span);
15                }
16            }
17        }
18        self.walk_expr(expr);
19        ControlFlow::Continue(())
20    }
21}
